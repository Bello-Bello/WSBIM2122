pkgs <- c("AnnotationDbi", "base", "Biobase", "BiocGenerics",
          "biomaRt", "datasets", "DelayedArray", "DESeq2", "dplyr", "DT",
          "forcats", "GenomeInfoDb", "GenomicRanges", "ggplot2", "graphics",
          "grDevices", "IRanges", "matrixStats", "methods", "org.Hs.eg.db",
          "parallel", "patchwork", "purrr", "readr", "S4Vectors", "stats",
          "stats4", "stringr", "SummarizedExperiment", "tibble", "tidyr",
          "tidyverse", "utils")
if (!require("BiocManager"))
   install.packages("BiocManager")
